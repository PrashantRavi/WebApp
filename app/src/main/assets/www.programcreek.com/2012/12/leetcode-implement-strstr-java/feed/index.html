return -1;
}

int j=0;
int nextIndex=-1;
boolean flag = false;
int k = haystack.length() - needle.length();
for(int i=0; i<=k; ){
int m=i;
for(j=0; j 0 && haystack.charAt(m) == needle.charAt(0) && !flag){
nextIndex = m;
flag = true;
}
m++;
}else{
break;
}

}

if(j == needle.length()){
return i;
}

if(flag){
i = nextIndex;
flag = false;
}else{
if(j!=0)
i = i+j;
else
i = i+1;
}
}

return -1;
}

}
}
