public class Solution {
public double findMedianSortedArrays(int[] nums1, int[] nums2) {
int le=nums1.length+nums2.length;
int l =(le)/2;

int[] arr =new int[l+1];
int j=0;
int k=0;
for(int i=0;i=nums1.length || k>=nums2.length){

if(j>=nums1.length){
arr[i]=nums2[k];
k++;
}else{
arr[i]=nums1[j];
j++;
}
}else{
if(nums1[j]nums2[k]){
arr[i]=nums2[k];
k++;
}else{
arr[i]=nums2[k];
if(i+1<=l){
i++;
arr[i]=nums1[j];
j++;
}

k++;
}
}
}
if(le%2==0){
return (arr[l]+arr[l-1])/2.0;
}else
return arr[l];
}
}
