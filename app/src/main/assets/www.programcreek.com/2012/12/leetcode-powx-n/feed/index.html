static double pow(double x, int n) {
if (n == 0) return 1;
double half = pow(x, n / 2);
if (n % 2 == 0) return half * half;
else if (n > 0) return half * half * x;
else return half * half / x;
}

public int pow(int x, int n) {
int result = 1;

while (n > 0) {
if (n % 2 != 0) {
result = result * x;
}

x = x * x;
n = n / 2;
}

return result;
}
