public class Solution {
public int removeDuplicates(int[] A) {

HashSet mySet = new HashSet();

for (int i=0; i<A.length;i++) mySet.add(A[i]) ;

int j = 0 ;

for(int i : mySet) {

A[j] = i;

j++;

}
for(int i=j; i<A.length;i++) A[i] = Integer.MAX_VALUE;

Arrays.sort(A) ;

return mySet.size();

}

}
