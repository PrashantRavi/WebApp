public static void main(String[] a){
int[] num={7,8,1,4,5};
int count=1;
for(int i=0;i<num.length;i++){
int j=i+1;
if(j<=2 && i<2){
if(num[i] < num[j]){
count++;
if(count == 3){
break;
}
}else{
count=0;
break;
}
}
}

System.out.println("true");
}else{
System.out.println("false");
}
}
