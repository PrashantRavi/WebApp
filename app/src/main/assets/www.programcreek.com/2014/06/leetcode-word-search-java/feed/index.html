]]> By: Haomin http://www.programcreek.com/2014/06/leetcode-word-search-java/#comment-7616 Sun, 22 Nov 2015 00:59:00 +0000 http://www.programcreek.com/?p=14387#comment-7616 public boolean dfs(char[][] board, String word, int i, int j, int k){
int m = board.length;
int n = board[0].length;

if(i<0 || j=m || j>=n){
return false;
}

if(board[i][j] == word.charAt(k)){
board[i][j]=’#’;
if(k==word.length()-1){
return true;
}else if(dfs(board, word, i-1, j, k+1)
||dfs(board, word, i+1, j, k+1)
||dfs(board, word, i, j-1, k+1)
||dfs(board, word, i, j+1, k+1)){
return true;
}
board[i][j]=word.charAt(k);
}

return false;
}
