public void connect(TreeLinkNode root) {

if(root==null)return;

TreeLinkNode left=root.left;

TreeLinkNode right=root.right;

if(left != null){

left.next = right;

if(root.next!=null)right.next = root.next.left;

}

connect(root.left);

connect(root.right);

}

}
