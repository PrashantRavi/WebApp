class MinStack {
Node head;

public void push(int x) {
if(head == null) {
head = new Node(x);
head.min = x;
}
else {
Node temp = new Node(x);
if(temp.min > head.min) temp.min = head.min;
temp.next = head;
head = temp;
}
}

public void pop() {
if(head == null ) return ;
head = head.next;
}

public int top() {
if(head == null ) return Integer.MAX_VALUE;
return head.value;
}

public int getMin() {
if(head == null ) return Integer.MAX_VALUE;
return head.min;
}
}

class Node {
int value;
int min;
Node next;
Node(int x) { value = x; next = null; min = x;}
}

min[index] = index == 0 ? x : Math.min(x, min[index-1]);
