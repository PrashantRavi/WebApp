public static int findStart(int[] gas, int[] cost) {

if (gas.length != cost.length) {

return -1;

}

int remaining = 0, start = 0;

for (int i = 0; i < gas.length; i++) {

remaining += gas[i] – cost[i];

if (remaining < 0) {

start = i + 1;

remaining = 0;

}

}

return start;

}
