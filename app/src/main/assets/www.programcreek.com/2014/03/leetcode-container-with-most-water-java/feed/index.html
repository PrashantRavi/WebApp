//heights: int[] array

function maxArea( heights ) {
var width = heights.length-1;
var objects = [];
for(var i = 0; i < width; i++){
var obj = {i:i,height:heights[i]};
objects.push(obj);
}
var sorted = objects.sort(function(a,b){
return (b.height*b.i – a.height*a.i)
});

return sorted[0]
}
