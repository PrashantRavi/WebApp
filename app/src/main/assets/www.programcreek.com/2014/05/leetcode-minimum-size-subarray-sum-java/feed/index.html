public int minSubArrayLen(int s, int[] nums) {
if (nums == null || nums.length == 0 || s <= 0) return 0;
int min = Integer.MAX_VALUE;
int i = 0, sum = 0;
for (int j = 0; j = s) {
min = Math.min(min, j – i + 1);
sum -= nums[i++];
}
}
return min == Integer.MAX_VALUE ? 0 : min;
}
