ArrayList list = new ArrayList();
for (int i : nums1) {
if(Arrays.binarySearch(nums2, i)!=-1){
list.add(i);
}
}

int[] result = new int[list.size()];
int i = 0;
while (i < list.size()) {
result[i] = list.get(i);
System.out.println(list.get(i) + ",");
i++;
}

return result;
}
