/**
* Definition for singly-linked list.
* public class ListNode {
* int val;
* ListNode next;
* ListNode(int x) { val = x; }
* }
*/
public class Solution {
public ListNode removeNthFromEnd(ListNode head, int n) {
if (head == null) return head;
ListNode p = head;
int len = 0;
while (p != null) {
len++;
if (p.next != null) {
p = p.next;
} else {
break;
}
}
// delete head case
if (n == len) return head.next;
ListNode end = p;
p.next = head;
int k = len – n;
while (k > 0) {
k–;
p = p.next;
}
end.next = null;
// delete tail case
if (n == 1) {
p.next = null;
} else {
p.next = p.next.next;
}
return head;
}
}
