public static int[] plusOne(int[] digits) {
int len = digits.length;
for (int i = len – 1; i >= 0; i–) {
if (digits[i] == 9) {
digits[i] = 0;
} else {
digits[i] = digits[i] + 1;
break;
}

if (i == 0 && digits[i] == 0) {
int[] y = new int[len + 1];
y[0] = 1;
for (int j = 1; j <= len; j++) {
y[j] = digits[j – 1];
}
digits = y;
}
}

return digits;
}

if (digits[i] == 9) {
digits[i] = 0;
} else {
++digits[i];
return digits;
}
}
//we have to add a digit at the head

for (int j = 1; j <= len; ++j) {
y[j] = digits[j – 1];
