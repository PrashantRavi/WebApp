public void moveZeros(int[] arr) {
if (arr.length < 2) return;
int j = 0;
int i = 1;
while (i < arr.length) {
if (arr[i] != 0 && arr[j] == 0) {
arr[j++] = arr[i];
arr[i] = 0;
}
i++;
}
}
