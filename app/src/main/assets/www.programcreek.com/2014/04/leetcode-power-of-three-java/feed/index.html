// numbers are – 1,3,9,27,81,243,729,2187,6561…..
// pattern is -divide by 3 power 2(number of digits -1) should be divisble by 3 or 9

if(n == 1 || n==3 || n==9){
return true;
}
else{
var a = n / (Math.Pow(3, 2*(n.toString().length() – 1)));
if (a == 3 || a == 9)
{
return true;
}
return false;
}
