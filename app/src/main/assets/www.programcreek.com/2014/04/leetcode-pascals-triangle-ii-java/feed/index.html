One loop. No additional storage. Same time bounds but faster in practice.


public List getRow(int rowIndex) {
ArrayList result = new ArrayList();

long c = 1;
for (int j = 0; j <= rowIndex; j++) {
result.add((int) c);
c *= rowIndex - j;
c /= j + 1;
}
return result;
}
