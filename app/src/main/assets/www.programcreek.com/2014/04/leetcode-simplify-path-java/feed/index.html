public static String simplifyPath(String path){

String[] array = path.split(“/”);

StringBuilder builder = new StringBuilder();

for(String item : array){

if(!item.contains(“.”) && !item.isEmpty()){

builder.append(“/”);

builder.append(item);

}

}
return builder.toString();
}
