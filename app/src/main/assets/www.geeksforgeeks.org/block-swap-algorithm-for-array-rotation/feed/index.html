The following code stores the array the array to be swapped in another array.

 
#include<stdio.h>
#include<stdlib.h>

void rotateArray(int a[],int d,int n)
{
    d%=n;
    int i,j=0;
    int *b=(int *) calloc (d,sizeof(int));
    for(i=0;i<d;i++)
        b[i]=a[i];
    for(i=d;i<n;i++)
        a[i-d]=a[i];
    for(i=n-d;i<n;i++)
        a[i]=b[i-n+d];
}

int main()
{
    int n,a[]={1,2,3,4,5,6,7},i,d=7;
    n=sizeof(a)/sizeof(a[0]);
    if((d%n)>0)
        rotateArray(a,d,n);
    for(i=0;i<n;i++)
        printf("%d ",a[i]);
    return 0;
}
