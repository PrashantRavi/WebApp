#include

#define _ ios_base::sync_with_stdio(0);cin.tie(0);

using namespace std;

#define ll long long

int main()

{

int N;

scanf(“%d”, &N);

int dp[N+4][N+4];

int arr[N+4];

for(int i = 0; i < N; ++i) {

scanf("%d", &arr[i]);

}

for(int i = 0; i < N; ++i) {

for(int j = 0; j < N; ++j) {

dp[i][j] = 1e9;

}

dp[i][i] = arr[i];

}

for(int l = 2; l < N+1; ++l) {

for(int i = 0; i < N – l + 1; ++i) {

int j = i + l – 1;

for(int k = i; k < j; ++k) {

dp[i][j] = min(dp[i][j], dp[i][k] * dp[k+1][j]);

}

}

}

cout<< dp[0][N-1] << "n";

return 0;

}
