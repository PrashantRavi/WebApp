//considering calculation upto 10 imes.

public class Exponent {

public static double getExponent(double x, int i)
{
if(i == 0)
return 1; // as e^0 = 0;

if(i > 10)
return 1;

else
{
double y = getExponent(x, i+1);
return 1 + (x / i) * y;
}
}
public static void main(String[] args)
{
double exponent = getExponent(0, 1);

System.out.println(“The exponent is : ” + exponent);
}

}
