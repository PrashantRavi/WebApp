int parity(int n).
{
int i, j,k;
k=0;
i=0;
while(n!=0)
{
if(n>=pow(2, i))
{
if((pow(2, i)<=n) && (pow(2, i+1)>n)).
{
j=n-pow(2, i);
n=j;
k=k+1;
i=0;
}
else
i=i+1;
}
}
return(k);
}

0x55555555 – 01010101 01010101 01010101 01010101
0x33333333 – 00110011 00110011 00110011 00110011
0x0F0F0F0F – 00001111 00001111 00001111 00001111
