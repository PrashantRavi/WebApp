But there is a recursive solution to this as well as written here – http://k2code.blogspot.in/2010/04/return-nth-node-from-end-of-linked-list.html:

node* findNthNode (node* head, int find, int& found){
if(!head) {
found = 1;
return 0;
}
node* retval = findNthNode(head->next, find, found);
if(found==find)
retval = head;
found = found + 1;
return retval;
}
