int kadane (int a[], int n)
{
int max_so_far = 0, max_ending_here = 0;
int i;
for(i = 0; i < 2 * n; i++)
{
max_ending_here = max_ending_here + a[i % n];
if(max_ending_here < 0)
max_ending_here = 0;
if(max_so_far < max_ending_here)
max_so_far = max_ending_here;
}
return max_so_far;
}

public static int getMaxSum(int[] a){

int n = a.length;

if(n==0) return 0;

if(n==1) return a[0];

int[] res = new int[2];

//res[0] is the final return and res[1] is the tmp sum in the first round

getMaxSum(a, 0, n-1, res);

getMaxSum(a, 0, n-2, res);

return res[0];

}

private static void getMaxSum(int[] a, int s, int e, int[] res){

int tmp = res[1];

for(int i=s; i<=e; i++){

tmp += a[i];

res[1] = tmp;

if(tmp<0){

tmp = 0;

}else{

res[0] = Math.max(res[0], tmp);

}

}

}
