public static boolean isLucky(int n){
int currentStepNumber = 2;
while(true){
if(n<currentStepNumber)
return true;
else if (n%currentStepNumber ==0)
return false;
else{
// In every step positions that are multiple of currentStepNumber are removed
// There will (n/currentStepNumber) such numbers that get deleted
// So new position of n is n – (n/currentStepNumber)
n = n – (n/currentStepNumber);
}

currentStepNumber++;
}
}
