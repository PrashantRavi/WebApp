public static int addAllGreaterKeys(Node node,int value)
{
if(node == null)
return 0;

if(node.right == null && node.left == null)
{
node.n += value;
return node.n;
}

int leftSum = addAllGreaterKeys(node.left, rightSum + node.n);

node.n += rightSum;

return leftSum;
}
