k = 8
a = [0,0,5,1,3,3,3,7,2,2,2]

for i in range(len(a)):
a[a[i]%k] += k

print a.index(max(a))

j = -1
for i in range(len(a)):
j = max(j, a[i]/k)
a[i] = a[i]%k

print j
