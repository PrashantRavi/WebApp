btnode * nextRight(btnode *t,int level,int k,int &kLevel)
{
if(t!=NULL)
{ if(!kLevel)
{
if(t->data==k)
{ kLevel=level;
return NULL;
}
}
else if(kLevel==level)
{ return t;
}

btnode *temp1=nextRight(t->lchild,level+1,k,kLevel);
if(temp1)
return temp1;

btnode *temp2=nextRight(t->rchild,level+1,k,kLevel);
return temp2;
}
else
{
return NULL;
}
}
