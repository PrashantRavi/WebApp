]]> By: Akash Panda http://www.geeksforgeeks.org/move-last-element-to-front-of-a-given-linked-list/#comment-34465 Tue, 08 Apr 2014 10:15:00 +0000 http://geeksforgeeks.org/?p=6850#comment-34465 void MoveLastToFront(struct node **head)

{

struct node *current=*head;

if(current==NULL || current->next==NULL)

return;

while(current->next->next!=NULL)

{

current=current->next;

}

struct node *temp=current->next;

current->next=NULL;

temp->next=*head;

*head=temp;

}
