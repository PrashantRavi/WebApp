#include

int main(char **argv, int argc)
{
int arr1[] = {10, 20};
int arr2[] = {10, 20};
int arr3[] = {10, 20};
int *p, value;

p = arr1;
value = ++*p;
printf(“arr1[0] = %d, arr1[1] = %d, *p = %d value = %dn”, arr1[0], arr1[1], *p, value);

p = arr2;
value = *p++;
printf(“arr2[0] = %d, arr2[1] = %d, *p = %d value = %dn”, arr2[0], arr2[1], *p, value);

p = arr3;
value = *++p;
printf(“arr3[0] = %d, arr3[1] = %d, *p = %d value = %dn”, arr3[0], arr3[1], *p, value);

return 0;
}

Output:
arr1[0] = 11, arr1[1] = 20, *p = 11 value = 11
arr2[0] = 10, arr2[1] = 20, *p = 20 value = 10
arr3[0] = 10, arr3[1] = 20, *p = 20 value = 20
