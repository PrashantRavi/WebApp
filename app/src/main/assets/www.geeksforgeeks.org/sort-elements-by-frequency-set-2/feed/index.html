import java.util.HashMap;

import java.util.LinkedList;

class data implements Comparable

{

Integer num ;

Integer frequency;

public data(int n , int f)

{

num = n;

frequency=f;

}

@Override

public int compareTo(Object a1) {

return this.frequency.compareTo(((data)a1).frequency);

}

}

public class FrequencySort {

public static void main(String args[])

{

int [] arr = {1,1,1,4,5,5,18,18,1,2,5,9,3,3,6,6,7,10};

HashMap map1 = new HashMap();

LinkedList l1 = new LinkedList();

int frequency = 1 ;

for(int x : arr)

{

map1.put(x, map1.get(x)==null?frequency:map1.get(x)+1);

}

for(int x : map1.keySet())

{

l1.add(new data(x,map1.get(x)));

}

Collections.sort(l1);

for(data x : l1)

{

System.out.println(x.num);

}

}

}
