void deleteNode(struct node **nodeptr){

int data;

struct node *temp , *nodePtr = *nodeptr;

if(nodePtr -> next == NULL){

*nodeptr = NULL;

}else{

data = nodePtr -> next -> data;

temp = nodePtr -> next;

nodePtr -> next = nodePtr -> next -> next;

nodePtr -> data = data;

free(temp);

}

}
