int isSubset(int a[],int b[],int sizeA,int sizeB)
{
int j=0;
int i;
int count = 0;
if(sizeA >=sizeB)
{
for(i=0;i<sizeA;i++)
{
if(b[j]==a[i])
{
count++;
j++;
i=0;
}
if(j==sizeB)
break;
}
if(count == sizeB)
printf("b[] is subset of a[]n");
else
printf("b[] is not a subset of a[]n");
}
