int leafLevel(btnode *t,int level)
{ if(t==NULL)
return 0;

if(t->lchild && t->rchild)
{
int k1=leafLevel(t->lchild,level+1);
int k2=leafLevel(t->rchild,level+1);
if(k1==0 && k2==0 || k1!=k2)
return 0;

return k1;
}
else if(!t->lchild && t->rchild)
{ return leafLevel(t->rchild,level+1);
}
else if (!t->rchild && t->lchild)
{ return leafLevel(t->lchild,level+1);
}
else
{ return level;
}
}
