#define n 7
#define max 20
int adj[max][max];
int te[max][max];
int be[max][max];
int status[max];
int dtime[max];
int ftime[max];
#define initial 0
#define active 1
#define backtracked 2
#define size 50
using namespace std;
void dfs(int v,int &time)
{
status[v]=active;
dtime[v]=++time;
for(int i=0;i<n;i++)
{
if(adj[v][i]==1)
{
if(status[i]==initial)
{
te[v][i]=1;
dfs(i,time);
}
if((status[i]==active)&&(te[i][v]!=1))
{
be[v][i]=1;
}

}
}
time++;
ftime[v]=time;
status[v]=backtracked;
}
void get_ap(int vi)
{
int child,flag;
for(int i=0;i<n,i!=vi;i++)
{
cout<<i;
for(int j=0;j<n;j++)
{
if(te[i][j]==1)
{
child=j;
cout<<child<<"is child of"<<i<<"n";
flag=0;
for(int k=0;k=dtime[child])&&(ftime[k]<=ftime[child]))
{
cout<<k<<"n";
for(int p=0;p<n;p++)
{
if((be[k][p]==1)&&(dtime[p]<dtime[i]))
{
flag=1;
}
}
}
}
if(!flag)
cout<<" vertex "<<i <<"is articulation point.n";
}
}
}
}
int main()
{
int time=0;int v1,v2,vi;
for(int i=0;i<n;i++)
{
dtime[i]=0;
status[i]=initial;
ftime[i]=0;
}
int m=n*(n-1)/2;
for(int i=0;i<n;i++)
{
for(int j=0;j<n;j++)
{
adj[i][j]=0;
te[i][j]=0;
be[i][j]=0;
}
}
for(int i=0;i<m;i++)
{
cout<>v1>>v2;
if((v1==-1)&&(v2==-1))
break;
adj[v1][v2]=1;
adj[v2][v1]=1;
}
cout<>vi;
dfs(vi,time);
get_ap(vi);
return 0;

}
