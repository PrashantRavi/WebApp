#include

void printKMax(int arr[], int n, int k)
{
int i, j, max;

max = arr[0]; //Initialize max

//Find Max element of first window i,e first sub-array of size k

for(i=1; imax)
{
max=arr[i];
}
}

//Print max element of first window i,e first sub-array of size k

printf(“n %d, “, max);

//Now compare ith element of array (which is also the last element of current window i,e sub-array) with max element of previous window (previous sub-array)
//if greater than max element of previous window than current element be the max element for current window else previous max element will also be the max element for current window
for(;imax)
{
max=arr[i];
}
printf(” %d, “, max);
}

}

int main(void)
{
int arr[10] = {8, 5, 10, 7, 9, 4, 15, 12, 90, 13};
int n=10, k = 4;
printKMax(arr, n, k);
return;
}

Output :

10 10 10 15 15 90 90
