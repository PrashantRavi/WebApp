static void numberFromRootToLeaf(Node root, int numTillNow,
List nums) {

if (root == null)
return;

numTillNow = (numTillNow * 10) + root.val;

if (root.left == null && root.right == null) {
nums.add(numTillNow);
return;
}

numberFromRootToLeaf(root.left, numTillNow, nums);
numberFromRootToLeaf(root.right, numTillNow, nums);
}

}
