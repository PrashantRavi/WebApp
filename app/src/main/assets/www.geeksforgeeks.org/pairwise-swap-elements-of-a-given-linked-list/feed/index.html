{

node *tmp;
node *prev = NULL;
if (!(*head))
return;
if (!(*head)->next)
return;
else
*head = (*head)->next;

{
tmp = current->next;
current->next = current->next->next;
tmp->next = current;

prev->next = tmp;
prev = current;
current = current->next;
}
