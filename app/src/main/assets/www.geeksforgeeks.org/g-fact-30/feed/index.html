]]> By: Sobhan Jachuck http://www.geeksforgeeks.org/g-fact-30/#comment-16116 Sat, 09 Mar 2013 11:59:18 +0000 http://geeksforgeeks.org/?p=8539#comment-16116 #include<iostream>
using namespace std;

class A{
int a;

public:
int *ptr;
A(){
ptr = new int[5];
}

~A(){
delete ptr;
}
};

int main(){

// new and free work fine But destructor is not called.
// Dynamically created array[5] is not deleted, which leads to memory leak
A *b = new A;
free(b);

// malloc and delete crash with STATUS_ACCESS_VIOLATION exception
// malloc does not call constructor so ptr has garbage value
// delete tries to call the destructor, which in turn tries to delete the location pointed by ptr.
// ptr might contain an address beyond the resident memory of the process, leading to STATUS_ACCESS_VIOLATION exception
A *a= (A *)malloc(sizeof(A));
cout<<(int)a->ptr<<endl; // ptr points to an arbitrary location, which means (ptr!=NULL)
delete a;

}
