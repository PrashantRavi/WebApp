void LCS(char str1[],char str2[], int result[]) {

int len1=strlen(str1),len2=strlen(str2);

char temp[2][len2];

int z=0;

for(int i=0;i<len1;i++){

for(int j=0;j=z){

z=temp[i%2][j];

result[0]=i-z+1;

result[1]=i;

}

}

else

temp[i%2][j]=0;

}

}

}

public class LCS {

public String LongestCommonSubsequence(String x, String y, int i, int j) {

String s = “”;

if (i == x.length() || j == y.length()) {

return “”;

}

if (x.charAt(i) == y.charAt(j)) {

s += x.charAt(i) + LongestCommonSubsequence(x, y, i + 1, j + 1);

return s;

} else {

String s1 = LongestCommonSubsequence(x, y, i + 1, j);

String s2 = LongestCommonSubsequence(x, y, i, j + 1);

return s1.length() < s2.length() ? s2 : s1;

}

}

public static void main(String[] args) {

LCS l = new LCS();

String x = "ABCDBAC";

String y = "BCDBDAB";

String s = l.LongestCommonSubsequence(x, y, 0, 0);

System.out.println(s);

}

}
