#include <iostream>
#include <ctime>
using namespace std;

int random(int a, int b) {
  static int first_run=1;
  if(a==0 && b==1) {
    if(first_run==1) {
      srand( (unsigned)time( NULL ) );
      first_run=0;
    }
    return rand()%2;
  } else if(a==b){
    return a;
  } else if(b==a+1) {
    if(random(0, 1)) {
      return b;
    } else {
      return a;
    }
  } else {
    if(random(0, 1)) {
      return random((a+b)/2+1, b);
    } else {
      return random(a, (a+b)/2);
    }
  }
}
