]]> By: gautam http://www.geeksforgeeks.org/searching-for-patterns-set-1-naive-pattern-searching/#comment-20800 Fri, 28 Jun 2013 05:52:29 +0000 http://geeksforgeeks.org/?p=11871#comment-20800 public HashSet<Integer> naivePatternSearch(String pattern, String string) {
HashSet<Integer> index = new HashSet<Integer>();
for (int j = 0; j < string.length(); j++) {
for (int i = 0; i < pattern.length(); i++) {
if (pattern.charAt(i) == string.charAt(j+i)) {
if(i==pattern.length()-1){
index.add(j);
}
} else {
break;
}
}
}

}
