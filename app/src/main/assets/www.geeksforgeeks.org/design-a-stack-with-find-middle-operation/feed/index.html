Algorithm:
1)if the count is 0 then there are no elements so return -1
2)
a)if count is 1, we need to update mid and head as NULL and decrement count
b)else
we update the mid->prev’s next and mid->next’s prev pointers provided they exist (for case where count = 0, mid->next would be null). if the count had been odd, we update mid as mid->next else mid->prev. finally we decrement count and delete the mid.

int deleteMiddle(struct myStack *ms){
if(ms->count==0)
return -1;
if(ms->count==1){
int data=ms->mid->data;
struct DLLNode * temp=ms->mid;
ms->mid=NULL;
ms->head=NULL;
free(temp);
ms->count-=1;
return data;
}
else{
struct DLLNode * temp=ms->mid;
int data=temp->data;
if(temp->prev)
temp->prev->next=temp->next;
if(temp->next)
temp->next->prev=temp->prev;
if(ms->count & 1){
ms->mid=temp->next;
}
else{
ms->mid=temp->prev;
}
ms->count -=1;
free(temp);
return data;
}
}
