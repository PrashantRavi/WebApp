long double area(int x1, int y1, int x2, int y2, int x3, int y3) {
   long double X= (x1*(y2-y3) + x2*(y3-y1)+ x3*(y1-y2))/2.0;
   if(X<0)return X*-1;
   return X;
}

   long double A2 = area (x1, y1, x, y, x3, y3);
   long double A3 = area (x1, y1, x2, y2, x, y);
