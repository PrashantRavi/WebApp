void search(char *pat, char *txt)
{
    int M = strlen(pat);
    int N = strlen(txt);
    int i = 0;  //points to location in text
    int j =0; //points to location for pattern
    for( int i = 0 ; i <= N-M ;i++)
    {
        if( txt[i] == pat[j] ) {
                j++;
        }else {
                j=0;
        }
        if( j == M ) {
                printf("Pattern found at index %d \n", i-M);
                 j=0;
        }
   }
 
}
