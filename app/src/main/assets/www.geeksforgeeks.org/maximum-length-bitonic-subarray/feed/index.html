#include
using namespace std;
int maxBiotonicSubArray(int a[], int n)
{
bool maxim_found = false;
int max_start, max_end, max_len;
int tmp_start, tmp_end;

int prev = a[0];
max_start = max_end = 0;
tmp_start = tmp_end = 0;

for (int i = 1; i prev)
{
if (maxim_found)// we have found the end of the current biotonic sub array
{
maxim_found = false;

if ( tmp_end – tmp_start + 1 > max_end – max_start + 1)
{
max_start = tmp_start;
max_end = tmp_end;

tmp_start = i – 1;
}
}
}
else
maxim_found = true;

tmp_end = i;
prev = a[i];
}

if ( tmp_end – tmp_start + 1 > max_end – max_start + 1)
{
max_start = tmp_start;
max_end = tmp_end;
}

for (int j = max_start; j <= max_end; j++)
cout << a[j] << " ";

cout << endl;

return max_end – max_start + 1;
}

int main()
{
int arr[] = {20, 4, 1, 2, 3, 4, 2, 10};//{12, 4, 78, 90, 45, 23};
int n = sizeof(arr)/sizeof(arr[0]);
cout << "Length of max length Bitnoic Subarray is " << maxBiotonicSubArray(arr, n) << endl;
return 0;
}
