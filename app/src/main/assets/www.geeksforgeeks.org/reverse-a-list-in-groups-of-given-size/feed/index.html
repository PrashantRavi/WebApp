f1 = current;

next = current.next;
current.next = prev;
prev = current;
current = next;

count++;
if (count == n) {
if (veryFirst) {
this.head = prev;
veryFirst = false;
}
count = 0;

f2.next = prev;
}
f2 = f1;
prev = null;
}
}

this.head = prev;
} else if (count != 0) { // if size of list mod n != 0…

}
