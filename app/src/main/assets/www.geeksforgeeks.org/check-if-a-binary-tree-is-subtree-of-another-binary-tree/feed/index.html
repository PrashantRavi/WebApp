using namespace std;

struct tree{
int data;
struct tree *lc,*rc;
};

tree *newNode(int n)
{
struct tree *node=new tree;
node->lc=node->rc=NULL;
node->data=n;

return node;
}

void check(struct tree *root1,struct tree *root2) throw(int)
{
if(root1 || root2)
{
check(root1->lc,root2->lc);
if( root1->data!=root2->data || root1==NULL || root2 ==NULL)
{
throw 1;
}

check(root1->rc,root2->rc);
}
}

void Traverse(struct tree *root1,struct tree *root2)throw (char)
{
if(root1 == NULL) return;

try{
if(root1->data == root2->data)
{
check(root1,root2);
throw ‘a’;
}
}

catch(int i){ }

Traverse(root1->lc,root2);
Traverse(root1->rc,root2);
}

int main()
{

struct tree *root1=NULL,*root2=NULL;

struct tree *T = newNode(26);
T->rc = newNode(3);
T->rc->rc = newNode(3);
T->lc = newNode(10);
T->lc->lc = newNode(4);
T->lc->lc->rc = newNode(30);
T->lc->rc = newNode(6);

struct tree *S = newNode(10);
S->rc = newNode(6);
S->lc = newNode(4);
S->lc->rc = newNode(30);

try {
Traverse(T,S);
cout<<"Not sub tree"<<endl;
}

catch(char a){ cout<<"subtree"<<endl; }
}
