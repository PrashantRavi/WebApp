using namespace std;
struct tnode
{
tnode* lchild;
int data;
tnode* rchild;
tnode(int d)
{
lchild=NULL;
data=d;
rchild=NULL;
}
};
tnode* left_rotate(tnode* root)
{
tnode* lptr,*temp;
lptr=root->rchild;
temp=lptr->lchild;
lptr->lchild=root;
root->rchild=temp;
return lptr;

}
tnode* right_rotate(tnode* root)
{
tnode* rptr,*temp;
rptr=root->lchild;
temp=rptr->rchild;
rptr->rchild=root;
root->lchild=temp;
return rptr;
}
tnode* splay(tnode* root,int d,int k)
{
if(root->data==d)
return root;
if(d>root->data)
root->rchild=splay(root->rchild,d,k);
if(ddata)
root->lchild=splay(root->lchild,d,k);
if((root->lchild)&&(root->lchild->data==d)&&(root->data==k))
{
return right_rotate(root);
}
if((root->rchild)&&(root->rchild->data==d)&&(root->data==k))
{
return left_rotate(root);
}
if((root->lchild)&&(root->lchild->lchild)&&(root->lchild->lchild->data==d))
{
root->lchild=right_rotate(root->lchild);
return right_rotate(root);
}
if((root->rchild)&&(root->rchild->rchild)&&(root->rchild->rchild->data==d))
{
root->rchild=left_rotate(root->rchild);
return left_rotate(root);
}
if((root->lchild)&&(root->lchild->rchild)&&(root->lchild->rchild->data==d))
{
root->lchild=left_rotate(root->lchild);
return right_rotate(root);
}
if((root->rchild)&&(root->rchild->lchild)&&(root->rchild->lchild->data==d))
{
root->rchild=right_rotate(root->rchild);
return left_rotate(root);
}
return root;

}

tnode* insert(tnode* root,int d)
{
if(!root)
{
root=new tnode(d);
return root;
}
if(d>root->data)
root->rchild=insert(root->rchild,d);
if(ddata)
root->lchild=insert(root->lchild,d);
return root;

}
