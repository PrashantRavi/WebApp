#include<stdio.h>

// Example of Function Overloading

void multiply(void *ptr, int s, int mul_with){
        int *ptr2Int;
        float *ptr2Float;

        if(s == 0){ 
                ptr2Int = (int *)ptr;
                (*ptr2Int) = (*ptr2Int) * (mul_with);
        }   
        else if(s==1) {
                ptr2Float = (float *)ptr;
                (*ptr2Float) = (*ptr2Float) * (mul_with);
        }   
}

int main() {

        int p = 10; 
        float q = 30.0;

        multiply(&p, 0, 10);    //Second argument 0 represents int and third argument is a constant to multiply with

        multiply(&q, 1, 20);    //Second argument 1 represents float

        printf("%d\n", p); 
        printf("%f\n", q); 
}
