void populateNext(struct node* p)
{
static struct node **next = NULL;

if (p)
{

populateNext(p->left);
if(next)
*next=p;
next=&p->next;
populateNext(p->right);
}
}

int isBST(struct node* node, struct node* prev)
{
if(!node)
{
return true;
}

if(!node->left)
{
isBST(node->left, prev);
}

if(!prev)
{
if(prev->data > node->data)
{
return false;
}
}

prev = node;

if(!node->right)
{
isBST(node->right, prev);
}
return true;
}
