#include

// A class with private destuctor

class Test

{

private:

~Test() {}

public:

void destructTest(Test* );

};

// Only this function can destruct objects of Test

void Test::destructTest(Test* ptr)

{

delete ptr;

}

int main()

{

// create an object

Test *ptr = new Test;

// destruct the object

ptr->destructTest(ptr);

return 0;

}
