private static void printAllPath(int[][] matrix, int r, int c,

int[] output, int index) {

// base case

if (r == ROW-1 && c == COL-1) {

output[index] = matrix[r];

printOutput(output, index);

return;

}

if (r >= ROW || c >= COL) {

return;

}

output[index] = matrix[r];

if (isSafe(r + 1, c)) {

printAllPath(matrix, r + 1, c, output, index + 1);

}

//printOutput(output,index);

if (isSafe(r, c + 1)) {

printAllPath(matrix, r, c + 1, output, index + 1);

}

//printOutput(output, index);

}

private static boolean isSafe(int i, int j) {

if (i < ROW && j < COL)

return true;

return false;

}

private static void printOutput(int[] output, int index) {

for (int i = 0; i <= index; i++) {

System.out.print(output[i] + ", ");

}

System.out.println();

}
