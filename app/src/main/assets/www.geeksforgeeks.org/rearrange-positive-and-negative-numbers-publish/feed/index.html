void swapWithPos(int[] arr, int i) {
for (int j = i + 1; j 0)
swap(arr, i, j);
return;
}

void swapWithNeg(int[] arr, int i) {
for (int j = i + 1; j < arr.length; j++) {
if (arr[j] < 0)
swap(arr, i, j);
}
return;
}

void swap(int[] arr, int i, int j) {
arr[i] = arr[i] + arr[j];
arr[j] = arr[i] – arr[j];
arr[i] = arr[i] – arr[j];
}
