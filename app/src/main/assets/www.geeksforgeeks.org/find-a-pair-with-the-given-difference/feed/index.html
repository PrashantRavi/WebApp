// The function to find the pair

bool findPair(int arr[], int size, int n)

{

int i = 0;

int maxElement = 0;

// Search for the maximum element

for(i = 0; i < size; i++) {

if(maxElement < arr[i])

maxElement = arr[i];

}

bool hashMap[maxElement + 1];

int tmp;

for(i = 0; i < maxElement + 1; i++) {

hashMap[i] = 0;

}

for(i = 0; i < size; i++) {

if(arr[i] < n) {

tmp = arr[i] + n;

if (tmp <= maxElement && hashMap[tmp]) {

printf("%d,%d",arr[i],tmp);

return;

}
}

else {

tmp = arr[i] – n;

if (tmp <= maxElement && hashMap[tmp]) {

printf("%d,%d",arr[i],tmp);

return;

}

}

hashMap[arr[i]] = 1;

}

printf("No such pair");

return false;

}

// Driver program to test above function

int main()

{

int arr[] = {5, 20, 3, 2, 50, 80};

int size = sizeof(arr)/sizeof(arr[0]);

int n = 78;

findPair(arr, size, n);

getchar();

return 0;

}

Time Complexity : O(n)
