int isSingleChildPreOrder(int a[], int n){

int i=0;

if(n==0 || n==1 || n==2)

return 1;

int left_min = INT_MIN;

int right_max = INT_MAX;

int left_max = a[0]-1;

int right_min = a[0]+1;

for(i=1;i<n;i++)
{
if(a[i]=left_min){
//left child

right_max = a[i-1]-1;
}

else if(a[i]>=right_min&&a[i]<=right_max){
//right child

left_min = a[i-1]+1;
}

else return 0;

left_max = a[i]-1;
right_min = a[i]+1;

}

return 1;

}
