So the actual result is = (9^3-9^2*3-9^1*2-1)-1=385 ….

int n;
string num;

inline int totdigits(int digit) { return (digit >= 3 ? digit : digit+1); }

int count3(int pos)
{
    int digit = num[pos] - '0';

    if (pos == n-1) {
        return totdigits(digit);
    }
    return (digit != 3) * count3(pos+1) + totdigits(digit-1) * pow(9, n-pos-1);
}
