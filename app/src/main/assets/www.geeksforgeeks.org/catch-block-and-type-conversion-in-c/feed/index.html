using namespace std;

class MyExcept1 {};

class MyExcept2
{
public:

    // Conversion constructor
    MyExcept2 (const MyExcept1 &e1 )
    {
        cout << "Conversion constructor called" << endl;
    }

{
    MyExcept1 e1;
    fun(e1);
    return 0;
}
