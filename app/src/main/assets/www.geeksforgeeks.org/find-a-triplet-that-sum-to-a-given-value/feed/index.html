void findTriplet(int arr[], int n, int sum) {
std::map triplet;
int flag = 0;
for(int i = 0; i < n; i++) {
for(int j = i+1; j < n; j++) {
triplet[arr[j]] = false;
}
for(int j = i+1; j < n; j++) {
if(triplet[sum-arr[i]-arr[j]] == true) {
std::cout << "(" << arr[i] << "," << arr[j] << "," << sum-arr[i]-arr[j] << ")";
flag = 1;
goto outer;
}
triplet[arr[j]] = true;
}
triplet[arr[i]] = true;
}
outer:
if(flag == 0) {
std::cout << "No triplet found !! ";
}
std::cout << std::endl;
}
