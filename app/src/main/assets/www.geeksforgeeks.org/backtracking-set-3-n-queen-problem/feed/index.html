using namespace std;

bool nQueens(int solve[],int n)

{

int j,r1,r2,c1,c2;

if(n==N)

return true;

for(int i=0;i<N;i++)

{

r2=n;

c2=i;

for(j=0;j<n;j++)

{ r1=j;

c1=solve[j];

if(r1==r2||c1==c2||abs(r1-r2)==abs(c1-c2))

break;

}

if(j==n)

{

solve[n]=i;

if(nQueens(solve,n+1))

return true;

}

}
return false;
}
