void Arr::weightedRandom(vector<int> a, vector<float> prob) {.
vector<float> ceil;
ceil.push_back(0);
for (int i=1; I < a.size(); i++) {.
ceil.push_back(prob[i]*100 + ceil[i-1]);.
}

vector<int> sum;
for (int i=0; I <= a.size(); i++) {.
sum.push_back(0);
}

for (int i=0; I < 10000; i++) {.
int rIndex = genRandomNo(ceil);.
sum[rIndex]++;

}

for (int i=0; I < sum.size(); i++) {.
cout << "npercent[ " << I << "]" << "=" << sum.at(i)/10000.0;.
}
}

int Arr::genRandomNo(vector<float> &ceil) {.
int r = rand() % 100;.
binSearchRange(ceil, 1, ceil.size(), r);.
return ceil.size();
}

int Arr::binSearchRange(vector<float> ceil, int l, int r, int v) {.
if (l > r).
return -1;

int mid = (l + r) / 2;.

if (v >= ceil[mid] && v <= ceil[mid+1]) {.
return mid;
}

if (v < ceil[mid]).
return binSearchRange(ceil, l, mid-1, v);.
if (v > ceil[mid+1] ).
return binSearchRange(ceil, mid+1, r, v);.
}
