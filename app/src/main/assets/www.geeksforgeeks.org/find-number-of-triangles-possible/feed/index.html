]]> By: aspire http://www.geeksforgeeks.org/find-number-of-triangles-possible/#comment-22331 Thu, 18 Jul 2013 15:36:22 +0000 http://www.cdn.geeksforgeeks.org/?p=24842#comment-22331 @GeeksforGeeks : I would like to suggest an O(n^2*logn) algorithm.
This algorithm runs two loops and then a binary search to find the first such element whose value is lesser than the sum of the 2 loop elements.

 
#include<stdio.h>
#include<stdlib.h>

int comp(const void *a,const void *b)
{
    return (*(int*)a-*(int*)b);
}

int binaryLastLesser(int a[],int l,int h,int sum)
{
    int m = (l+h)/2;
    if(l>h)
        return -1;
    else
    {
        if(a[m]>=sum)
            return binaryLastLesser(a,l,m-1,sum);
        else        ///a[m]<sum
        {
            if(m+1>h||((m+1<=h)&&a[m+1]>=sum))
                return m;
            else
                return binaryLastLesser(a,m+1,h,sum);
        }
    }
}

int noOfTriangles(int a[],int n)
{
    int i,j,cnt=0,index;
    qsort(a,n,sizeof(int),comp);
    for(i=0;i<=n-3;i++)
    {
        for(j=i+1;j<=n-2;j++)
        {
            index = binaryLastLesser(a,j+1,n-1,a[i]+a[j]);
            if(index!=-1)
            {
                printf("%d %d %d\n",a[i],a[j],a[index]);
                cnt++;
            }
        }

    }
    return cnt;
}

int main()
{
    int a[]= {3,3,7},n;
    n=sizeof(a)/sizeof(a[0]);
    printf("The number of triangles in the array are: %d\n",noOfTriangles(a,n));
    return 0;
}
