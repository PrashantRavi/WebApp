                (((s1[i] >= 'A' && s1[i] <= 'Z') || (s1[i] >= 'a' && s1[i] <= 'z')) &&

           continue;
        else

    if (s1[i] == s2[i])

  if (a==b) { printf("Both the strings are same\n"); return 0;}
  else if(!(tolower(*a)>=65 && tolower(*a)<=90) && !(tolower(*a)>=97 && tolower(*a)<=122) && tolower(*a)!='\0') {printf("ERROR: %s is not an alphabetical string\n", a); return -2;}
