for( i=1; i <= square ; i++ ){
magic[ row ][ col ] = i;
if( i % n == 0 ) row++;
else{
if( row == 0 ) row = n - 1;
else row--;
if( col == ( n - 1 ) ) col = 0;
else col++;
}
}
