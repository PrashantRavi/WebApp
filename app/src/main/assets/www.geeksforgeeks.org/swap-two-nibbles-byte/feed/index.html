The concept mentioned above is basic concept so that anyone can apply it character or integer. The output will be expected , but if we remove ‘&’ and 0xFF… , then we cannot change the bytes in case of integer. Lets see an example –

=====================

#include
void SwapNibbles(unsigned char *);
void HTONS(unsigned int *);
int main()
{
unsigned char uCnumber;
uCnumber = 100;
unsigned int address = 167772161;

/* Operation for Character */

printf(“%xn”, uCnumber);
SwapNibbles(&uCnumber);
printf(“%xn”, uCnumber);

/* Operation for Integer */
printf(“%xn”,address);
HTONS(&address);
printf(“%xn”,address);

}
/* With ‘&’ and 0X..FF..*/

void SwapNibbles(unsigned char *uCnumber)
{
*uCnumber =(((*uCnumber & 0XF0) >>4 ) | ((*uCnumber & 0X0F) <>4 ) | (*uCnumber <> 24) | ((*address& 0X00FF0000) >> 8) | ((*address& 0X000000FF) << 24) | ((*address& 0X0000FF00) <> 24) | (*address >> 8) | (*address << 24) | (*address <>4 ) | (*uCnumber <> 24) | (*address >> 8) | (*address << 24) | (*address << 8));
}
