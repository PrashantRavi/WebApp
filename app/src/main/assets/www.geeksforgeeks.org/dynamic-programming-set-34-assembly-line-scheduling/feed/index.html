int fun(int i,int j,int a[][4],int t[][4],int x[],int arr[][4],int high)
{
if(arr[i][j]!=-1)
return arr[i][j];
if(j==high)
{
if(i==0)
return x[0]+a[i][j];
else
return x[1]+a[i][j];
}
arr[i][j+1]=fun(i,j+1,a,t,x,arr,high);
if(i==0)
{
arr[i+1][j+1]=fun(i+1,j+1,a,t,x,arr,high);
arr[i][j]=a[i][j]+min(arr[i][j+1],t[i][j+1]+arr[i+1][j+1]);
return arr[i][j];
}
else
{
arr[i-1][j+1]=fun(i-1,j+1,a,t,x,arr,high);
arr[i][j]=a[i][j]+min(arr[i][j+1],t[i][j+1]+arr[i-1][j+1]);
return arr[i][j];
}
}
