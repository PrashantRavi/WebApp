I tested it with a few random cases and it works fine. If you find any mistake,please comment.

 
double myPow(int m,int n){
	if(n==0)
		return 1;
	bool isNeg=( n<0 )? true : false;
	n=abs(n);
	double ans=m;
	for(int i=2;i<=n;++i)
		ans=multiply(ans,m);
	if(isNeg)
		return 1/ans;
	return ans;
}

int multiply(int x,int y){
	if(y==0)
		return 0;
	if(y>0)
		return x+multiply(x,y-1);
	return -multiply(x,-y);
}
