]]> By: samthebest http://www.geeksforgeeks.org/tree-isomorphism-problem/#comment-38119 Tue, 10 Jun 2014 15:39:00 +0000 http://www.cdn.geeksforgeeks.org/?p=117828#comment-38119 int isIsomorphic(btnode *t1,btnode *t2)
{
if(t1!=NULL && t2!=NULL)
{
if(t1->data!=t2->data)
{ return 0;
}

if(!isIsomorphic(t1->lchild,t2->lchild))
{ return isIsomorphic(t1->lchild,t2->rchild) && isIsomorphic(t1->rchild,t2->lchild);
}

return isIsomorphic(t1->rchild,t2->rchild);
}
else if (!t1 && !t2)
{ return 1;
}
else
{ return 0;
}
}
