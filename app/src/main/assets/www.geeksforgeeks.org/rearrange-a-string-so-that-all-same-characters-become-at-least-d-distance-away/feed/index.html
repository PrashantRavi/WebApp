{

int i,j,n=0,l=0,k=0;

/* code for calculate each character and it’s frequency */

for(i = 0 ; i < size ; i++)
{
for(j = 0 ; j < n ; j++)
{
if(arr[j].c == str[i]) // if character is already store then increase it's count only
{
arr[j].count++;
break;
}
}

if( j == n ) // if character is not store then store it and it's frequency 1
{
arr[n].c=str[i];
arr[n++].count=1;
}
str[i]=''; // store null in string when character is used for future use of str
}

/* store character in struct node array in decreasing order frequency */

// bubble sort for sorting

struct node temp;
for( i = 0 ; i < n ; i++ )
