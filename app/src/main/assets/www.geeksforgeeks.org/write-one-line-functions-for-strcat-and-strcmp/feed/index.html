//Logic used
bool my_strcmp(char *a, char *b)
{
	if (*a == *b)
	{
		if (*a == '\0')
		{
			return true;
		} else {
			my_strcmp(++a, ++b);
		}
	} else {
		return false;
	}
} 

// Desired One line function
bool my_strcmp(char *a, char *b)
{
	return ((*a == *b) ? ((*a == '\0') ? true : my_strcmp(++a, ++b)): false);
}
