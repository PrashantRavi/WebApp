]]> By: Ernesto http://www.geeksforgeeks.org/add-1-to-a-given-number/#comment-20182 Thu, 20 Jun 2013 12:57:08 +0000 http://geeksforgeeks.org/?p=8198#comment-20182 I found the solution for the edge case without comparing with -1:

 
int plus_one_recursive(int x) {
    if (x & 1 == 0) return x | 1;
    if (x >> 1 == x) return 0;
    return plusOne(x >> 1) << 1;
}

int plus_one_iterative(int x) {
    if (x != 0 && x >> 1 == x) return 0;
    int accum = 1;
    while ((x & 1) != 0) {
        x >>= 1;
        accum <<= 1;
    }
    x |= 1;
    while (accum > 1) {
        x <<= 1;
        accum >>= 1;
    }
    return x;
}
