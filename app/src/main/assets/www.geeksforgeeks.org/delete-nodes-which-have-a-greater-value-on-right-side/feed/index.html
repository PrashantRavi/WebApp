struct node {

int data;

struct node *next;

};

struct node *remove(struct node *prev, struct node *current)

{

static int max = 0;

if (current->next == NULL) {

max = current->data;

return current;

}

remove(current, current->next);

if (current->data >= max)

max = current->data;

else {

if (prev != NULL) {

prev->next = current->next;

delete current;

return prev;

}

else {

struct node *tmp = current->next;

delete current;

return tmp;

}

}

return current;

}
