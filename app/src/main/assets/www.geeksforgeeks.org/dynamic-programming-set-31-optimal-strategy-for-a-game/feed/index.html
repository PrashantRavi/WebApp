if(right – left == 1) {
dp[left][right] = max(arr[left],arr[right]);
return max(arr[left],arr[right]);
}

if(dp[left][right] > 0 && left ret2 + arr[right]) {
dp[left][right] = ret1 + arr[left];
return dp[left][right];
} else {
dp[left][right] = ret2 + arr[right];
return dp[left][right];
}

} else {
dp[left][right] = min(ret2,ret1);
return min(ret2,ret1);
}
}
