{

int max_value = Price[n – 1];
//base
if (n == 1)
{
OP[0] = Price[0];
return Price[0];
}
//Just go thro half the length, eg for length=7 as(3+4) is the same as (4+3)
int toLength = n / 2;
for (int i = 0; i < toLength; i++){
if (OP[i] == 0) cutRod(Price, ((n – 1) – i), OP);
max_value = Math.Max(max_value, OP[i] + OP[(n – 2) – i]);//more efficient as you dont have to calculate the sbuoptimal soulutions as we store it
}
OP[n – 1] = max_value;//store the suboptimal solutions
return max_value;
}
