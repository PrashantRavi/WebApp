    cout<<"\nEnter the pattern: ";

    pattern_matching_naive(str, pattern,strlen(str), strlen(pattern)); 
    getch();
    return 0;
}

void pattern_matching_naive(char *str, char *pattern, int n, int k)
{
    int i,j, pos;
    for(i = 0; i < n; i++)
    {
        pos = i;
        for(j=0; j<k; )
        {
            if (pattern[j] == str[i])
            {
                j++;
                i++;
            }
            else
                break;
        }
        if (j == k)
        {
            cout<<"\nPattern found at i = \n" <<pos;
            i--;
        } 
        else
            i = i - j;      
    }
}
