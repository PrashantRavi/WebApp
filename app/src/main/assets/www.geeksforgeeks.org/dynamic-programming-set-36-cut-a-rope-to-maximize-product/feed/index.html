#define ll long long

using namespace std;

ll maxproduct(ll n)
{
if(n==0 || n==1)
return 0;

if(n==2)
return 1;

if(n==3)
return 2;

if(n%3==0)
return pow(3.0,n/3);

if(n%3==1)
return pow(3.0,(n-4)/3)*4;

if(n%3==2)
return pow(3.0,(n-2)/3)*2;

}

{
ll n;
cin>>n;
cout << "Maximum Product is " << maxproduct(n);

}
