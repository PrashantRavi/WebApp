{
node *slowptr,*fastptr;
slowptr = fastptr = head;
if(head == NULL)
{
printf(“nnList is empty.”);
return FALSE;
}
while(slowptr && fastptr && fastptr->next)
{
fastptr = fastptr->next->next;
slowptr = slowptr->next;
if(slowptr == fastptr)
return TRUE;
}
return FALSE;
}
